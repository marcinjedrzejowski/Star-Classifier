    Dict("kernel" => "rbf", "degree" => 3, "C" => 0.1, "gamma" => "scale"),

    Dict("kernel" => "linear", "degree" => 5,  "C" => 0.1, "gamma" => "auto"),

    Dict("kernel" => "linear", "degree" => 3, "C" => 1.0, "gamma" => "auto"),
